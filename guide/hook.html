      # 基本参数

      DANGER

      • 类名必须和此文件名称
      • 或与上层文件夹名称相同(要求文件名称为index.php)

      编写好的插件建议放入extensive_plugin文件夹，因为plugin内的文件夹会随自动更新删除

      规范的参数如下 (打*为必须有的参数)

      参数 说明
      *name 插件的名字
      *profile 插件的简介说明
      *version 插件的版本
      *author 插件的作者
      *method 插件的请求方法
      *request_par 插件的请求参数及解释
      *return_par 插件的返回参数及解释
      type 插件的分类(无此项自动归为一些工具分类)

      支持MarkDown的项

      • profile
      • request_par
      • return_par

      基本格式

      DETAILS 
      # test.php
# `public`均可省略，省略默认为`public`,仅`private`需标明
# `$request`是可选参数，为网站传入的请求参数，为get或post数据，由 key `method`决定

<?php
class test{
    public function getInfo() {
        return [
            'name' => 'demo',
            'version' => '2023',
            'profile'=> 'demo0001',
            'method'=>'GET',
            'author'=>'molanp',
            'request_par'=> re_par(['a'=>'-']),
            'return_par'=> re_par(['data'=>'info']),
        ];
    }
    public function run($request) {
        // more...
        _return_($xxx);
    }
}

      # 示例插件

      # 目录结构
# 此文件夹置于`extensive_plugin`文件夹
# |name_generator
# |--|--index.php

<?php
class name_generator {
    function getInfo() {
        return [
            'name' => '亚文化取名机',
            'version' => '1.0',
            'profile'=> '一键生成最潮的亚文化网名',
            'method'=>'GET',
            'author'=>'molanp',
            'type'=>'潮',
            'request_par'=> re_par(['-'=>'-']),
            'return_par'=> re_par(['data'=>'你的亚文化名字']),
        ];
    }
    function run() {
        $el1 = ["废墟", "深海", ..,"社会主义", "无聊"];
        $el2 = ["小丑", "仿生", .., "堕落", "放射性"];
        $el3 = ["天使", "精灵", .., "博士", "糖"];
        _return_($el1[array_rand($el1)].$el2[array_rand($el2)].$el2[array_rand($el2)]);
    }
}

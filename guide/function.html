    # 网络请求方法

    # 一切以services/until.php文件及其注释为准

    # re_add

    重新格式化参数为 Markdown 表格

    # 参数

    • $type (array) - 请求方法数组
    • $url (array) - 请求 URL 和说明信息数组

    # 返回值

    • (string) - 格式化后的 Markdown 表格

    # 示例

    $type = ['GET', 'POST'];
$url = ['/api/get'=>'获取信息', '/api/post'=>'提交数据'];

$table = re_add($type, $url);
echo $table;

    # re_par

    重新格式化参数为 Markdown 表格

    # 参数

    • $key (array) - 参数键和参数值数组

    # 返回值

    • (string) - 格式化后的 Markdown 表格

    # 示例

    $key = ['id'=>'用户id', 'name'=>'用户名'];

$table = re_par($key);
echo $table;

    # _return_

    返回结果

    # 参数

    • $content (mixed) - 返回的数据内容
    • $status (int) - 状态码，可选，默认为 200

    # 返回值

    • 无

    # 示例

    $content = ['message' => '请求成功'];
$status = 200;
$location = false;

_return_($context, $status, $location);

    # handle_check

    检查 API 状态

    # 参数

    • $name (string) 插件名称

    # 返回值

    • (bool) - 是否需要处理 API 请求

    # 示例

    if (handle_check($name)) {
    // 处理 API 请求
} else {
    // 不处理 API 请求
}

    # find_files

    递归查询文件

    # 参数

    • $dir (string) - 目录路径
    • $file (string) - 需要查询的文件名，支持匹配文件末尾字符查询，默认为.php
    • $prefix (string) - 前缀，可选，默认为空字符串

    # 返回值

    • (array) - 文件的相对路径数组

    # 示例

    $dir = '/path/to/directory';
$prefix = '';
$file = '.php';
# 查找$dir内的.php结尾的文件
$files = find_files($dir, $file, $prefix);
print_r($files);

    # json

    尝试将 JSON 字符串解析为数组，或将数组转换为 JSON 字符串

    # 参数

    • $json (string|array) - 要解析的 JSON 字符串或要转换的数组

    # 返回值

    • (array|string) - 解析后的数组或数组转换后的 JSON 字符串，如果解码和编码都失败则返回原始字符串

    # 示例

    $json = '{"message": "Hello, world!"}';

$result = json($json);
print_r($result);

    # RequestLimit

    检查请求次数是否超过限制

    [插件开发者无需阅读此句]此函数应在 handle_check() 后执行

    # 参数

    • $limit (string) - 最大请求次数和时间单位，例如 '3/s' '10/min', '5/hour', '100/day'

    # 返回值

    • 若请求超过设定上限，自动结束该脚本

    # addSlashIfNeeded

    在给定的字符串开头或末尾添加斜杠("/")，如果它尚未以斜杠结尾。

    # 参数

    • $inputString (string) 要处理的字符串

    # 返回值

    • (string) 处理后的字符串，确保以斜杠结尾

    # 示例

    $a = addSlashIfNeeded('/example');
echo $a;
# /example/

    # logger 类

    一个用于日志记录的类，支持 info、warn、debug 和 error 四种级别的日志记录。

    # 方法

    # info($str)

    记录 info 级别的日志。

    • 参数:
      • $str (string) - 要记录的日志内容

    # warn($str)

    记录 warn 级别的日志。

    • 参数:
      • $str (string) - 要记录的日志内容

    # debug($str)

    记录 debug 级别的日志。

    • 参数:
      • $str (string) - 要记录的日志内容

    # error($str)

    记录 error 级别的日志。

    • 参数:
      • $str (string) - 要记录的日志内容

    # 示例

    $log = new logger();
$log->info("这是一条 info 级别的日志");
$log->warn("这是一条 warn 级别的日志");
$log->debug("这是一条 debug 级别的日志");
$log->error("这是一条 error 级别的日志");
